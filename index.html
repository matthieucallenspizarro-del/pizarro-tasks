// ============================================================ // PIZARRO TASK DASHBOARD — Google Apps Script Backend v2 // All communication via GET + URL parameters (CORS-safe) // // Deploy > New deployment > Web App // Execute as: Me | Who has access: Anyone // ============================================================ var SHEET_TASKS = "Tasks"; var SHEET_PROJECTS = "Projects"; var SHEET_MEMBERS = "Members"; function doGet(e) { // Add CORS headers via ContentService var action = e.parameter.action; var result; try { if (action === "init") result = initSheets(); else if (action === "getTasks") result = getTasks(); else if (action === "getProjects") result = getProjects(); else if (action === "getMembers") result = getMembers(); else if (action === "saveTask") result = saveTask(e.parameter); else if (action === "deleteTask") result = deleteTask(e.parameter.id); else if (action === "saveProject") result = saveProject(e.parameter); else if (action === "saveMember") result = saveMember(e.parameter); else if (action === "deleteMember") result = deleteMember(e.parameter.id); else if (action === "deleteProject") result = deleteProject(e.parameter.id); else result = { ok: true, status: "Pizarro Backend v2 running. Use ?action=init to set up sheets." }; } catch(err) { result = { error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── INIT ──────────────────────────────────────────────────── function initSheets() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var taskSheet = ss.getSheetByName(SHEET_TASKS); if (!taskSheet) { taskSheet = ss.insertSheet(SHEET_TASKS); taskSheet.appendRow(["id","proj","name","desc","status","deadline","assignee"]); styleHeader(taskSheet, 7); } var projSheet = ss.getSheetByName(SHEET_PROJECTS); if (!projSheet) { projSheet = ss.insertSheet(SHEET_PROJECTS); projSheet.appendRow(["id","name","color"]); styleHeader(projSheet, 3); projSheet.appendRow([0, "Shared Tasks", "#C8102E"]); projSheet.appendRow([1, "Personal — Matthieu", "#2563eb"]); projSheet.appendRow([2, "Personal — Dimitri", "#16a34a"]); } var memSheet = ss.getSheetByName(SHEET_MEMBERS); if (!memSheet) { memSheet = ss.insertSheet(SHEET_MEMBERS); memSheet.appendRow(["id","name","color","cat"]); styleHeader(memSheet, 4); memSheet.appendRow(["M", "Matthieu", "#C8102E", "overhead"]); memSheet.appendRow(["D", "Dimitri", "#2563eb", "overhead"]); } return { ok: true, message: "Sheets ready" }; } function styleHeader(sheet, cols) { sheet.getRange(1, 1, 1, cols) .setFontWeight("bold") .setBackground("#C8102E") .setFontColor("#ffffff"); } // ── TASKS ─────────────────────────────────────────────────── function getTasks() { var rows = getRows(SHEET_TASKS); return { tasks: rows }; } function saveTask(p) { var sheet = getOrCreate(SHEET_TASKS, ["id","proj","name","desc","status","deadline","assignee"]); var data = sheet.getDataRange().getValues(); var headers = data[0]; // Decode fields (URL params are strings) var task = { id: p.id || "", proj: p.proj || "0", name: decodeURIComponent(p.name || ""), desc: decodeURIComponent(p.desc || ""), status: p.status || "in-progress", deadline: p.deadline || "", assignee: p.assignee || "" }; // Try to update existing row for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === String(task.id)) { var row = headers.map(function(h) { return task[h] !== undefined ? task[h] : ""; }); sheet.getRange(i + 1, 1, 1, row.length).setValues([row]); return { ok: true, action: "updated", id: task.id }; } } // New task var newId = String(Date.now()); task.id = newId; var newRow = headers.map(function(h) { return task[h] !== undefined ? task[h] : ""; }); sheet.appendRow(newRow); return { ok: true, action: "created", id: newId }; } function deleteTask(id) { return deleteRow(SHEET_TASKS, id); } // ── PROJECTS ──────────────────────────────────────────────── function getProjects() { return { projects: getRows(SHEET_PROJECTS) }; } function saveProject(p) { var sheet = getOrCreate(SHEET_PROJECTS, ["id","name","color"]); var data = sheet.getDataRange().getValues(); var maxId = 0; data.slice(1).forEach(function(r) { if (parseInt(r[0]) > maxId) maxId = parseInt(r[0]); }); var newId = maxId + 1; sheet.appendRow([newId, decodeURIComponent(p.name || ""), p.color || "#C8102E"]); return { ok: true, id: newId }; } // ── MEMBERS ───────────────────────────────────────────────── function getMembers() { return { members: getRows(SHEET_MEMBERS) }; } function saveMember(p) { var sheet = getOrCreate(SHEET_MEMBERS, ["id","name","color","cat"]); sheet.appendRow([p.id, decodeURIComponent(p.name || ""), p.color || "#C8102E", p.cat || "overhead"]); return { ok: true }; } function deleteMember(id) { return deleteRow(SHEET_MEMBERS, id); } // ── HELPERS ───────────────────────────────────────────────── function getRows(sheetName) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) return []; var data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; var headers = data[0]; return data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); } function getOrCreate(name, headers) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); styleHeader(sheet, headers.length); } return sheet; } function deleteRow(sheetName, id) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) return { ok: false, error: "Sheet not found" }; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === String(id)) { sheet.deleteRow(i + 1); return { ok: true }; } } return { ok: false, error: "Row not found" }; } // ── DELETE PROJECT (add to doGet handler) ────────────────── // Add this case to the if/else chain in handleRequest: // else if (action === "deleteProject") result = deleteProject(e.parameter.id); function deleteProject(id) { // Remove from Projects sheet deleteRow(SHEET_PROJECTS, id); // Also remove all tasks belonging to this project var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_TASKS); if (!sheet) return { ok: true }; var data = sheet.getDataRange().getValues(); // Delete from bottom up to avoid row shift issues for (var i = data.length - 1; i >= 1; i--) { if (String(data[i][1]) === String(id)) { sheet.deleteRow(i + 1); } } return { ok: true }; }
